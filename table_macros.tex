%%-----------------------------------------------------------------------------
%% Helpful Custom Macros for Managing Tables
%%-----------------------------------------------------------------------------

%%-----------------------------------------------------------------------------
%% Required packages
%%-----------------------------------------------------------------------------
% Enables nicer fixed with column times
\usepackage{array}
% Enable 'sub tables' for cleaner milestones table
\usepackage{multirow}

%%-----------------------------------------------------------------------------
%% Standard/Package Formatting Value Adjustments
%%-----------------------------------------------------------------------------
% Shrink space between captions and figure
\newcommand{\tabcapspaceshrink}{{\vspace{-18pt}}}
\newcommand{\tabprespaceshrink}{{\vspace{-18pt}}}
\newcommand{\tabpostpaceshrink}{{\vspace{-18pt}}}

%%-----------------------------------------------------------------------------
%% Custom Figure Placement Macros
%%-----------------------------------------------------------------------------
% Normal Table placement
% (inputs the tabular section from the provided file)
% Usage: \puttab{filename}{label}{Caption}
\newcommand{\puttab}[3]{%
\tabprespaceshrink
\begin{table}[!htb]
\caption{#3}
\label{tab:#2}
\tabcapspaceshrink
\begin{singlespace}
\begin{center}
\input{Tables/#1}
\end{center}
\end{singlespace}
\end{table}
\tabpostpaceshrink
}

% Table placement as a Horizontal page
% (inputs the tabular section from the provided file)
% Usage: \puttabh{filename}{label}{Caption}
\newlength{\twidth}
\newlength{\theight}

\newcommand{\puttabh}[3]{%
\setlength{\twidth}{\textwidth}
\setlength{\theight}{\textheight}

\begin{sidewaystable}
  % The following two lines compensate for what I think is a bug.
  \setlength{\textwidth}{\theight}
  \setlength{\textheight}{\twidth}
  \caption{#3}
  \label{tab:#2}
  \tabcapspaceshrink
  \begin{singlespace}
  \begin{center}
  \hbox to \textwidth{
    \input{Tables/#1}
  }
  \end{center}
  \end{singlespace}
\end{sidewaystable}
}

\newcolumntype{R}[1]{>{\raggedleft\let\newline\\\arraybackslash\hspace{0pt}}m{#1}}
